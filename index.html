From Idea to API in 3 Steps

kind: "Workflow"
metadata:
  name: "customer_onboarding"

steps:
  - id: "validate"
    kind: "functional"
    runner: "validate_customer"

  - id: "ai_scoring"
    kind: "agent"
    agent:
      model: "ollama/llama3"
      prompt: "Score this customer..."
    routes:
      premium: "fast_track"
      standard: "normal_flow"
from tuvl.nodes import node
from tuvl.repositories import get_repository

@node("validate_customer")
async def validate_customer(ctx: dict) -> dict:
    """Validate and enrich customer data."""
    session = ctx["_session"]
    repo = get_repository("Customer", session)
    
    customer = await repo.add({
        "email": ctx["email"],
        "name": ctx["name"],
    })
    
    return {**ctx, "customer_id": customer.id}
# Your workflow is now an API!

curl -X POST http://localhost:8000/workflows/customer_onboarding \
  -H "Content-Type: application/json" \
  -d '{
    "email": "alice@example.com",
    "name": "Alice Smith"
  }'
