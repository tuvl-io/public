# workflows/contact_intake.yaml
kind: "Workflow"
metadata:
  name: "contact_intake"
  description: "Capture and prioritize new contacts"

context: "Contact"

trigger:
  path: "/api/contacts"
  method: "POST"
  input_schema: "context"
  response_schema: "context"

steps:
  - id: "validate"
    kind: "functional"
    runner: "validate_contact"
    routes:
      valid: "save"
      invalid: "END"

  - id: "save"
    kind: "functional"
    runner: "save_contact"
    routes:
      default: "notify"

  - id: "notify"
    kind: "functional"
    runner: "send_welcome_email"
# nodes/contact_nodes.py
from typing import Any
from tuvl_engine.nodes.base import node
from tuvl_engine.repositories.registry import get_repository

@node("validate_contact")
async def validate_contact(ctx: dict[str, Any]) -> str:
    """Return signal based on validation result."""
    if not ctx.get("email") or "@" not in ctx["email"]:
        return "invalid"
    return "valid"

@node("save_contact")
async def save_contact(ctx: dict[str, Any]) -> dict[str, Any]:
    """Persist contact to the database."""
    session = ctx["_session"]
    repo = get_repository("Contact", session)
    contact = await repo.add({
        "email": ctx["email"],
        "name":  ctx["name"],
    })
    ctx["id"] = str(contact.id)
    return ctx

@node("send_welcome_email")
async def send_welcome_email(ctx: dict[str, Any]) -> dict[str, Any]:
    """Dispatch welcome notification."""
    ctx["notified"] = True
    return ctx

# workflows/onboarding.yaml
kind: "Workflow"
metadata:
  name: "candidate_onboarding"
  description: "AI-powered candidate screening"

context: "Candidate"

trigger:
  path: "/api/candidates/onboard"
  method: "POST"

steps:
  - id: "save_draft"
    kind: "functional"
    runner: "save_candidate"
    routes:
      default: "ai_vetting"

  - id: "ai_vetting"               # ✦ Agent step
    kind: "agent"
    agent:
      model: "ollama/llama3"
      system: "You are a fair hiring assistant."
      prompt: |
        Evaluate this candidate:
        Name: {{ full_name }}
        Experience: {{ experience_years }} years

        Classify as senior (5+ yrs), needs_review,
        or unqualified (<1 yr).
        Return JSON: {"decision": "..."}
      output:
        format: json
        map:
          decision: vetting_decision
        signal_from: decision
      retry:
        attempts: 3
        on: [parse_error, timeout]
    routes:
      senior:       "fast_track"
      needs_review: "manual_review"
      unqualified:  "END"

  - id: "fast_track"
    kind: "functional"
    runner: "fast_track_candidate"

  - id: "manual_review"
    kind: "functional"
    runner: "flag_for_review"
# nodes/onboarding.py
from typing import Any
from tuvl_engine.nodes.base import node
from tuvl_engine.repositories.registry import get_repository

@node("save_candidate")
async def save_candidate(ctx: dict[str, Any]) -> dict[str, Any]:
    """Persist candidate record before AI evaluation."""
    session = ctx["_session"]
    repo = get_repository("Candidate", session)
    candidate = await repo.add({
        "email":            ctx["email"],
        "full_name":        ctx["full_name"],
        "experience_years": ctx.get("experience_years", 0),
        "status":           "pending",
    })
    ctx["id"]     = str(candidate.id)
    ctx["status"] = candidate.status
    return ctx  # ↓ ai_vetting step runs next

@node("fast_track_candidate")
async def fast_track_candidate(ctx: dict[str, Any]) -> dict[str, Any]:
    """Mark senior candidates for immediate interview."""
    session = ctx["_session"]
    repo = get_repository("Candidate", session)
    await repo.update(ctx["id"], {"status": "fast_tracked"})
    ctx["status"] = "fast_tracked"
    return ctx

@node("flag_for_review")
async def flag_for_review(ctx: dict[str, Any]) -> dict[str, Any]:
    """Queue candidate for human assessment."""
    session = ctx["_session"]
    repo = get_repository("Candidate", session)
    await repo.update(ctx["id"], {"status": "review"})
    ctx["status"] = "review"
    return ctx

# models/contact.yaml
kind: "ModelDefinition"
version: "v1"
metadata:
  name: "Contact"
spec:
  tablename: "contacts"
  schema: true
  fields:
    - name: "id"
      type: "uuid"
      primary_key: true
      default: "uuid4"
    - name: "email"
      type: "string"
      unique: true
      required: true
    - name: "name"
      type: "string"
      required: true
    - name: "priority"
      type: "string"
      input: false
      description: "AI-assigned priority level"
# workflows/contact_intake.yaml
kind: "Workflow"
context: "Contact"
trigger:
  path: "/api/contacts"
  method: "POST"

steps:
  - id: "save"
    kind: "functional"
    runner: "save_contact"

  - id: "prioritize"        # ✦ LLM step
    kind: "agent"
    agent:
      model: "ollama/llama3"
      prompt: |
        Contact: {{ name }} — {{ email }}
        Assign a priority: high | medium | low
        Return JSON: {"priority": "..."}
      output:
        format: json
        map:
          priority: priority
    routes:
      default: "enrich"

  - id: "enrich"
    kind: "functional"
    runner: "enrich_contact"
